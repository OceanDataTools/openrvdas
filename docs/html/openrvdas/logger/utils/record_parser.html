openrvdas.logger.utils.record_parser

Tools for parsing NMEA and other text records.

By default, will load device and device_type definitions from files in logger/devices/.yaml and contrib/devices/.yaml. Please see documentation in contrib/devices/README.md for a description of the format these definitions should take.

  4
  5By default, will load device and device_type definitions from files in
  6logger/devices/*.yaml and contrib/devices/*.yaml. Please see documentation
  7in contrib/devices/README.md for a description of the format these
  8definitions should take.

 37class RecordParser:
 38    ############################
 39    def __init__(self, record_format=None,
 40                 field_patterns=None, metadata=None,
 41                 definition_path=DEFAULT_DEFINITION_PATH,
 42                 return_das_record=False, return_json=False,
 43                 metadata_interval=None, strip_unprintable=False,
 44                 quiet=False, prepend_data_id=False, delimiter=':'):
 45        """Create a parser that will parse field values out of a text record
 46        and return either a Python dict of data_id, timestamp and fields,
 47        a JSON encoding of that dict, or a binary DASRecord.
 48        ```
 49        record_format - string for parse.parse() to use to break out data_id
 50            and timestamp from the rest of the message. By default this will
 51            look for 'data_id timestamp field_string', where 'field_string'
 52            is a str containing the fields to be parsed.
 53
 54        field_patterns
 55            If not None, a list of parse patterns to be tried instead
 56            of looking for device definitions along the definition path,
 57            or a dict of message_type:[parse_pattern, parse_pattern].
 58
 59        metadata
 60            If field_patterns is not None, the metadata to send along with
 61            data records.
 62
 63        definition_path - a comma-separated set of file globs in which to look
 64            for device and device_type definitions with which to parse message.
 65
 66        return_json - return the parsed fields as a JSON encoded dict
 67
 68        return_das_record - return the parsed fields as a DASRecord object
 69
 70        metadata_interval - if not None, include the description, units
 71            and other metadata pertaining to each field in the returned
 72            record if those data haven't been returned in the last
 73            metadata_interval seconds.
 74
 75        strip_unprintable
 76                Strip out and ignore any leading or trailing non-printable binary
 77                characters in the string to be parsed.
 78
 79        quiet - if not False, don't complain when unable to parse a record.
 80
 81        prepend_data_id - If true prepend the instrument data_id to field_names
 82            in the record.
 83
 84        delimiter
 85            The string to insert between data_id and field_name when prepend_data_id is true.
 86            Defaults to ':'.
 87            Not used if prepend_data_id is false.
 88        ```
 89        """
 90        if not PARSE_INSTALLED:
 91            raise ImportError('RecordParser requires Python "parse" module; '
 92                              'please run "pip install parse"')
 93
 94        self.strip_unprintable = strip_unprintable
 95        self.quiet = quiet
 96        self.field_patterns = field_patterns
 97        self.metadata = metadata or {}
 98        self.record_format = record_format or DEFAULT_RECORD_FORMAT
 99        self.compiled_record_format = parse.compile(format=self.record_format,
100                                                    extra_types=extra_format_types)
101        self.return_das_record = return_das_record
102        self.return_json = return_json
103        if return_das_record and return_json:
104            raise ValueError('Only one of return_json and return_das_record '
105                             'may be true.')
106
107        self.metadata_interval = metadata_interval
108        self.metadata_last_sent = {}
109        self.prepend_data_id = prepend_data_id
110        self.delimiter = delimiter
111
112        # If we've been explicitly given the field_patterns we're to use for
113        # parsing, compile them now. Patterns may either be a list of strings,
114        # a dict of strings or a dict of lists of strings.
115        if field_patterns:
116            self.compiled_field_patterns = self._compile_formats_from_patterns(field_patterns)
117            self.metadata = metadata
118
119        # If we've not been given field_patterns to use for parsing, read in all
120        # the devices and device types to compile them.
121        else:
122            # Fill in the devices and device_types - NOTE: we won't be using
123            # these if 'field_patterns' is provided as an argument.
124            definitions = load_definitions(definition_path)
125            self.devices = definitions.get('devices', {})
126            self.device_types = definitions.get('device_types', {})
127
128            # Some limited error checking: make sure that all devices have a
129            # defined device_type.
130            for device, device_def in self.devices.items():
131                device_type = device_def.get('device_type')
132                if not device_type:
133                    raise ValueError('Device definition for "%s" has no declaration of '
134                                     'its device_type.' % device)
135                if device_type not in self.device_types:
136                    raise ValueError('Device type "%s" (declared in definition of "%s") '
137                                     'is undefined.' % (device_type, device))
138
139            # Compile format definitions so that we can run them more
140            # quickly. If format is a single string, normalize it into a list
141            # to simplify later code.
142            for device_type, device_type_def in self.device_types.items():
143                format = device_type_def.get('format')
144                if format is None:
145                    raise ValueError('Device type %s has no format definition'
146                                     % device_type)
147                compiled_format = self._compile_formats_from_patterns(format)
148                self.device_types[device_type]['compiled_format'] = compiled_format
149
150            # Metadata: If we haven't been handed a dict of metadata, compile it from
151            # the devices we've read.
152            #
153            # It's a map from variable name to the device and device type it
154            # came from, along with device type variable and its units and
155            # description, if provided in the device type
156            # definition. Compiling this information is kind of excruciating
157            # and voluminous.
158            if not metadata and metadata_interval is not None:
159                for device, device_def in self.devices.items():  # e.g. s330
160                    device_type_name = device_def.get('device_type')  # Seapath330
161                    if not device_type_name:
162                        raise ValueError('Device definition for "%s" has no declaration of '
163                                         'its device_type.' % device)
164                    device_type_def = self.device_types.get(device_type_name)
165                    if not device_type_def:
166                        raise ValueError('Device type "%s" (declared in definition of "%s")'
167                                         ' is undefined.' % (device_type_name, device))
168                    device_type_fields = device_type_def.get('fields')
169                    if not device_type_fields:
170                        raise ValueError('Device type "%s" has no fields?'
171                                         % device_type_name)
172
173                    fields = device_def.get('fields')
174                    if not fields:
175                        raise ValueError('Device "%s" has no fields?!?' % device)
176
177                    # e.g. device_type_field = GPSTime, device_field = S330GPSTime
178                    for device_type_field, device_field in fields.items():
179                        # e.g. GPSTime: {'units':..., 'description':...}
180                        field_desc = device_type_fields.get(device_type_field)
181                        if not field_desc:
182                            logging.warning('Device type "%s" has no field corresponding to '
183                                            'device field "%s"' % (device_type_name,
184                                                                   device_type_field))
185                            continue
186                        self.metadata[device_field] = {
187                            'device': device,
188                            'device_type': device_type_name,
189                            'device_type_field': device_type_field,
190                        }
191                        self.metadata[device_field].update(field_desc)

194    def parse_record(self, record):
195        """Parse an id-prefixed text record into a Python dict of data_id,
196        timestamp and fields.
197        """
198        if not record:
199            return None
200        if not isinstance(record, str):
201            if not self.quiet:
202                logging.info('Record is not a string: "%s"', record)
203            return None
204        try:
205            # Break record into (by default) data_id, timestamp and field_string
206            parsed_record = self.compiled_record_format.parse(record).named
207        except (ValueError, AttributeError):
208            if not self.quiet:
209                logging.warning('Unable to parse record into "%s"', self.record_format)
210                logging.warning('Record: %s', record)
211            return None
212
213        data_id = parsed_record.get('data_id', 'no_data_id')
214
215        # Convert timestamp to numeric, if it's there
216        timestamp = parsed_record.get('timestamp')
217        if timestamp is not None and isinstance(timestamp, datetime.datetime):
218            timestamp = timestamp.timestamp()
219            parsed_record['timestamp'] = timestamp
220
221        # Extract the field string we're going to parse; remove trailing
222        # whitespace.
223        field_string = parsed_record.get('field_string')
224
225        # If we don't have fields, there's nothing to parse
226        if field_string is None:
227            if not self.quiet:
228                logging.warning('No field_string found in record "%s"', record)
229            return None
230
231        if self.strip_unprintable:
232            field_string = ''.join([c for c in field_string if c.isprintable()])
233        field_string = field_string.strip()
234        if not field_string:
235            if not self.quiet:
236                logging.warning('No field_string found in record "%s"', record)
237            return None
238
239        # If we've been given a set of field_patterns: If they're a dict, see if there's
240        # a key that matches our data_id. If so, only look at those patterns. Otherwise
241        # try them all and use the first that matches.
242        if self.field_patterns:
243            # If field_patterns is a dict, see if our data_id matches any of the keys
244            if isinstance(self.field_patterns, dict):
245                patterns = self.compiled_field_patterns.get(data_id)
246            else:
247                patterns = self.compiled_field_patterns
248
249            # If no patterns to try, go home emptyhanded
250            if not patterns:
251                if not self.quiet:
252                    logging.warning(f'No parse field patterns matched data_id "{data_id}"')
253                return None
254            fields, message_type = self._parse_field_string(field_string, patterns)
255
256        # If we were given no explicit field_patterns to use, we need to
257        # count on the record having a data_id that lets us figure out
258        # which device, and therefore which field_patterns to try.
259        else:
260            fields, message_type = self.parse_for_data_id(data_id, field_string)
261
262        # We should now have a dictionary of fields. If not, go home
263        if fields is None:
264            if not self.quiet:
265                logging.warning('No formats matched field_string "%s"', field_string)
266            return None
267
268        # Some folks want the data_id prepended
269        if self.prepend_data_id:
270            # This conditional dictates whether fields are stored with just the
271            # field_name key, or <data_id><delimiter><field_name>
272            # Doing some work directly on the fields dict, so we'll take a copy
273            # to loop over.
274            fields_copy = fields.copy()
275            # Reset the fields dict
276            fields = {}
277            for field in fields_copy:
278                # Determine the new "field_name"
279                key = '' + data_id + self.delimiter + field
280                # Set the value
281                fields[key] = fields_copy[field]
282
283        # Remove raw 'field_string' and add parsed 'fields' to parsed_record
284        del parsed_record['field_string']
285        parsed_record['fields'] = fields
286        if message_type:
287            parsed_record['message_type'] = message_type
288
289        # Metadata Injection - use shared utility
290        metadata = collect_metadata_for_fields(
291            fields, timestamp, self.metadata,
292            self.metadata_interval, self.metadata_last_sent
293        )
294        if metadata:
295            parsed_record['metadata'] = metadata
296
297        logging.debug('Created parsed record: %s', pprint.pformat(parsed_record))
298
299        # What are we going to do with the result we've created?
300        if self.return_das_record:
301            try:
302                return DASRecord(data_id=data_id, timestamp=timestamp,
303                                 message_type=message_type, fields=fields,
304                                 metadata=metadata)
305            except KeyError:
306                return None
307
308        elif self.return_json:
309            return json.dumps(parsed_record)
310        else:
311            return parsed_record

314    def _parse_field_string(self, field_string, compiled_field_patterns):
315        # Default if we don't match anything
316        fields = message_type = None
317
318        # If our pattern(s) are just a single compiled parser, try parsing and
319        # return with no message type.
320        if isinstance(compiled_field_patterns, parse.Parser):
321            result = compiled_field_patterns.parse(field_string)
322            if result:
323                fields = result.named
324
325        # Else, if it's a list, try it out on all the elements.
326        elif isinstance(compiled_field_patterns, list):
327            for pattern in compiled_field_patterns:
328                fields, message_type = self._parse_field_string(field_string, pattern)
329                if fields is not None:
330                    break
331
332        # If it's a dict, try out on all values, using the key as message type.
333        # It's syntactically possible for the internal set of patterns to have
334        # their own message types. Not sure why someone would ever create patterns
335        # that did this, but if they do, let that override our base one.
336        elif isinstance(compiled_field_patterns, dict):
337            for message_type, pattern in compiled_field_patterns.items():
338                fields, int_message_type = self._parse_field_string(field_string, pattern)
339                message_type = int_message_type or message_type
340                if fields is not None:
341                    break
342        else:
343            raise ValueError('Unexpected pattern type in parser: %s'
344                             % type(compiled_field_patterns))
345
346        return fields, message_type

349    def parse_for_data_id(self, data_id, field_string):
350        """Look up the device and device type for a data_id. Parse the field_string
351        according to those formats. If successful, return a tuple of
352        (field_dict, message_type), where field_dict is a dict of
353        {field_name: field_value}. Return ({}, None) if unable to match a format pattern.
354        """
355        failure_values = (None, None)
356        if not self.devices:
357            logging.warning('RecordParser has no device definitions; unable to parse!')
358            return failure_values
359
360        # Get device and device_type definitions for data_id
361        device = self.devices.get(data_id)
362        if not device:
363            if not self.quiet:
364                logging.warning('Unrecognized data id "%s", field string: %s',
365                                data_id, field_string)
366                logging.warning('Known data ids are: "%s"',
367                                ', '.join(self.devices.keys()))
368            return failure_values
369
370        device_type = device.get('device_type')
371        if not device_type:
372            if not self.quiet:
373                logging.error('Internal error: No "device_type" for device %s!', device)
374            return failure_values
375
376        device_definition = self.device_types.get(device_type)
377        if not device_definition:
378            if not self.quiet:
379                logging.error('No definition found for device_type "%s"', device_type)
380            return failure_values
381
382        compiled_format_patterns = device_definition.get('compiled_format')
383        parsed_fields, message_type = \
384            self._parse_field_string(field_string, compiled_format_patterns)
385
386        # Did we get anything?
387        if parsed_fields is None:
388            if not self.quiet:
389                logging.warning('No formats matched field_string "%s"', field_string)
390            return failure_values
391
392        logging.debug('Got fields: %s', pprint.pformat(parsed_fields))
393
394        # Finally, convert field values to variable names specific to device
395        device_fields = device.get('fields')
396        if not device_fields:
397            if not self.quiet:
398                logging.error('No "fields" definition found for device %s', data_id)
399            return failure_values
400
401        # Assign field values to the appropriate named variable.
402        fields = {}
403        for field_name, value in parsed_fields.items():
404            variable_name = device_fields.get(field_name)
405            # None means we're not supposed to report it.
406            if variable_name is None:
407                continue
408            # None means we didn't have a value for this field; omit it.
409            if value is None:
410                continue
411            # If it's a datetime, convert to numeric timestamp
412            if isinstance(value, datetime.datetime):
413                value = value.timestamp()
414            fields[variable_name] = value
415
416        logging.debug('Got fields: %s', pprint.pformat(fields))
417        return fields, message_type

420    def _compile_formats_from_patterns(self, field_patterns):
421        """Return a list/dict of patterns compiled from the
422        str/list/dict of passed field_patterns.
423        """
424        if isinstance(field_patterns, str):
425            return [parse.compile(format=field_patterns,
426                                  extra_types=extra_format_types)]
427        elif isinstance(field_patterns, list):
428            return [parse.compile(format=p, extra_types=extra_format_types)
429                    for p in field_patterns]
430        elif isinstance(field_patterns, dict):
431            compiled_field_patterns = {}
432            for message_type, message_pattern in field_patterns.items():
433                compiled_patterns = self._compile_formats_from_patterns(message_pattern)
434                compiled_field_patterns[message_type] = compiled_patterns
435            return compiled_field_patterns
436
437        else:
438            raise ValueError('Passed field_patterns must be str, list or dict. Found %s: %s'
439                             % (type(field_patterns), str(field_patterns)))

241        # a key that matches our data_id. If so, only look at those patterns. Otherwise
242        # try them all and use the first that matches.
243        if self.field_patterns:

315    def _parse_field_string(self, field_string, compiled_field_patterns):
316        # Default if we don't match anything
317        fields = message_type = None
318
319        # If our pattern(s) are just a single compiled parser, try parsing and
320        # return with no message type.
321        if isinstance(compiled_field_patterns, parse.Parser):
322            result = compiled_field_patterns.parse(field_string)
323            if result:
324                fields = result.named
325
326        # Else, if it's a list, try it out on all the elements.
327        elif isinstance(compiled_field_patterns, list):
328            for pattern in compiled_field_patterns:
329                fields, message_type = self._parse_field_string(field_string, pattern)
330                if fields is not None:
331                    break
332
333        # If it's a dict, try out on all values, using the key as message type.
334        # It's syntactically possible for the internal set of patterns to have
335        # their own message types. Not sure why someone would ever create patterns
336        # that did this, but if they do, let that override our base one.
337        elif isinstance(compiled_field_patterns, dict):
338            for message_type, pattern in compiled_field_patterns.items():
339                fields, int_message_type = self._parse_field_string(field_string, pattern)
340                message_type = int_message_type or message_type
341                if fields is not None:
342                    break
343        else:
344            raise ValueError('Unexpected pattern type in parser: %s'
345                             % type(compiled_field_patterns))
346
347        return fields, message_type

350    def parse_for_data_id(self, data_id, field_string):
351        """Look up the device and device type for a data_id. Parse the field_string
352        according to those formats. If successful, return a tuple of
353        (field_dict, message_type), where field_dict is a dict of
354        {field_name: field_value}. Return ({}, None) if unable to match a format pattern.
355        """
356        failure_values = (None, None)
357        if not self.devices:
358            logging.warning('RecordParser has no device definitions; unable to parse!')
359            return failure_values
360
361        # Get device and device_type definitions for data_id
362        device = self.devices.get(data_id)
363        if not device:
364            if not self.quiet:
365                logging.warning('Unrecognized data id "%s", field string: %s',
366                                data_id, field_string)
367                logging.warning('Known data ids are: "%s"',
368                                ', '.join(self.devices.keys()))
369            return failure_values
370
371        device_type = device.get('device_type')
372        if not device_type:
373            if not self.quiet:
374                logging.error('Internal error: No "device_type" for device %s!', device)
375            return failure_values
376
377        device_definition = self.device_types.get(device_type)
378        if not device_definition:
379            if not self.quiet:
380                logging.error('No definition found for device_type "%s"', device_type)
381            return failure_values
382
383        compiled_format_patterns = device_definition.get('compiled_format')
384        parsed_fields, message_type = \
385            self._parse_field_string(field_string, compiled_format_patterns)
386
387        # Did we get anything?
388        if parsed_fields is None:
389            if not self.quiet:
390                logging.warning('No formats matched field_string "%s"', field_string)
391            return failure_values
392
393        logging.debug('Got fields: %s', pprint.pformat(parsed_fields))
394
395        # Finally, convert field values to variable names specific to device
396        device_fields = device.get('fields')
397        if not device_fields:
398            if not self.quiet:
399                logging.error('No "fields" definition found for device %s', data_id)
400            return failure_values
401
402        # Assign field values to the appropriate named variable.
403        fields = {}
404        for field_name, value in parsed_fields.items():
405            variable_name = device_fields.get(field_name)
406            # None means we're not supposed to report it.
407            if variable_name is None:
408                continue
409            # None means we didn't have a value for this field; omit it.
410            if value is None:
411                continue
412            # If it's a datetime, convert to numeric timestamp
413            if isinstance(value, datetime.datetime):
414                value = value.timestamp()
415            fields[variable_name] = value
416
417        logging.debug('Got fields: %s', pprint.pformat(fields))
418        return fields, message_type
