openrvdas.logger.transforms.nmea_transform

Take in a dict of various values and emit NMEA strings appropriate for them. NMEATransform is a thin wrapper around a set of individual NMEA message-generating transforms.

Each transform's __init__(self) method should expect a single 'kwargs' dict as its initialization argument, in which it will search for the keyword args it needs in order to function. If it does not find the necessary args, then rather than throwing an error, its transform method should just always return None.

Each transform's transform(record) method should expect its input in standard OpenRVDAS dict format:

{'timestamp':5345345, 'fields':{'field1': value1, 'field2':value2,...}}

Each transform should return a (possibly empty) list of NMEA strings.

If they're not too terribly ugly, the NMEA transforms should be placed in this file so as to minimize the risk that they may be inadvertently used elsewhere.

 40def checksum(source):
 41    """Return hex checksum for source string."""
 42    return '%02X' % reduce(xor, (ord(c) for c in source))

 46class NMEATransform(Transform):
 47    """Call our various component transforms and generate NMEA strings from them.
 48    """
 49
 50    def __init__(self, nmea_list: list = [], **kwargs):
 51        """
 52        nmea_list
 53                List of the nmea transforms that will be used.
 54        **kwargs
 55                Arugments needed for the nmea transforms, see transforms below for what will be included.
 56        """
 57        super().__init__(**kwargs)  # processes 'quiet' and type hints
 58
 59        self.transforms = []
 60
 61        # If nmea_list is not given as list, force it into one
 62        if not isinstance(nmea_list, list):
 63            nmea_list = [nmea_list]
 64
 65        if not nmea_list:
 66            self.transforms = [MWDTransform(kwargs), XDRTransform(kwargs)]
 67            return
 68
 69        class_module_name = 'logger.transforms.nmea_transform'
 70        module = importlib.import_module(class_module_name)
 71
 72        # Get all classes within this file
 73        classes = [cls_name for cls_name, cls_obj in inspect.getmembers(module) if
 74                   inspect.isclass(cls_obj)]
 75
 76        for transform in nmea_list:
 77            if transform in classes:
 78                class_const = getattr(module, transform)
 79                self.transforms.append(class_const(kwargs))
 80            else:
 81                logging.error('%s is not in classes %s', transform, classes)
 82
 83
 84    ############################
 85    def transform(self, record):
 86        """Expect a record dict (with 'timestamp' and 'fields' keys."""
 87        results = []
 88
 89        # Do we have more than one record here? Normalize so that
 90        # following code assumes a list of records.
 91        if not type(record) is list:
 92            record = [record]
 93
 94        for single_record in record:
 95            for t in self.transforms:
 96                result = t.transform(single_record)
 97                logging.debug('transform %s: %s', t, result)
 98
 99                # Transforms may return zero, one or more results
100                if not result:
101                    continue
102                elif type(result) is list:
103                    results.extend(result)
104                else:
105                    results.append(result)
106
107        # Just keep the results that are non-empty
108        pruned_results = [r for r in results if r]
109
110        # Return None, a single result or a list of results
111        if len(pruned_results) == 0:
112            return None
113        elif len(pruned_results) == 1:
114            return pruned_results[0]
115        else:
116            return pruned_results

120"""MWD - Wind Direction & Speed
121$--MWD, x.x,T,x.x,M,x.x,N,x.x,M*hh<CR><LF>
122
123$--: Talker identifier*
124MWD: Sentence formatter*
125x.x,T: Wind direction, 0° to 359° true*
126x.x,M: Wind direction, 0° to 359° magnetic*
127x.x,N: Wind speed, knots*
128x.x,M: Wind speed, meters/second*
129*hh: Checksum*
130
131We get true wind direction ab initio, but if we don't have access to
132vessel's magnetic variation, we can't generate the magnetic wind
133direction, so omit if not available.
134"""
135################################################################################
136
137
138class MWDTransform:
139    """Output a NMEA MWD string, given true wind and (when available)
140    magnetic variation.
141    """
142
143    def __init__(self, kwargs):
144        """
145        Look for these keys in the kwargs dict:
146        ```
147        true_wind_dir_field
148                 Field name to look for true wind direction
149        true_wind_speed_kt_field
150                 Field name to look for wind speed in knots. Either this
151                 or true_wind_speed_ms_field must be non-empty.
152        true_wind_speed_ms_field
153                 Field name to look for wind speed in meters per second.
154                 Either this or true_wind_speed_kt_field must be non-empty.
155        magnetic_variation_field
156                 Vessel magnetic variation. If omitted, only true winds
157                 will be emitted.
158        mwd_talker_id
159                 Should be format '--MWD' to identify the instrument
160                 that's creating the message.
161        ```
162        """
163        self.true_wind_dir_field = kwargs.get('true_wind_dir_field')
164        self.true_wind_speed_kt_field = kwargs.get('true_wind_speed_kt_field')
165        self.true_wind_speed_ms_field = kwargs.get('true_wind_speed_ms_field')
166        self.magnetic_variation_field = kwargs.get('magnetic_variation_field')
167        self.mwd_talker_id = kwargs.get('mwd_talker_id')
168
169        self.true_wind_dir = None
170        self.true_wind_speed_kt = None
171        self.true_wind_speed_ms = None
172        self.magnetic_variation = None
173
174    ############################
175    def transform(self, record):
176        """Incorporate any useable fields in this record. If it gives us a
177        new MWD record, return it.
178        """
179        # Check that we've got the right record type - it should be a
180        # single record.
181        if not record or type(record) is not dict:
182            logging.warning('Improper type for record: %s', type(record))
183            return None
184        fields = record.get('fields')
185        if not fields:
186            logging.debug('MWDTransform got record with no fields: %s', record)
187            return None
188
189        # Grab any relevant values
190        self.true_wind_dir = fields.get(self.true_wind_dir_field,
191                                        self.true_wind_dir)
192        if self.true_wind_speed_kt_field:
193            self.true_wind_speed_kt = fields.get(self.true_wind_speed_kt_field,
194                                                 self.true_wind_speed_kt)
195        if self.true_wind_speed_ms_field:
196            self.true_wind_speed_ms = fields.get(self.true_wind_speed_ms_field,
197                                                 self.true_wind_speed_ms)
198        if self.magnetic_variation_field:
199            self.magnetic_variation = fields.get(self.magnetic_variation_field,
200                                                 self.magnetic_variation)
201
202        # Do we have enough values to emit a record? If not, go home.
203        if self.true_wind_dir is None:
204            logging.debug('Not all required values present - skipping')
205            return None
206        if self.true_wind_speed_kt is None and self.true_wind_speed_ms is None:
207            logging.debug('Not all required values present - skipping')
208            return None
209
210        # Are we filling in meters per second from knots?
211        if self.true_wind_speed_ms_field is None and \
212           self.true_wind_speed_kt_field and \
213           self.true_wind_speed_kt is not None:
214            self.true_wind_speed_ms = self.true_wind_speed_kt * 0.514444
215
216        # Are we filling in knots from meters per second from?
217        if self.true_wind_speed_kt_field is None and \
218           self.true_wind_speed_ms_field and \
219           self.true_wind_speed_ms is not None:
220            self.true_wind_speed_kt = self.true_wind_speed_kt * 1.94384
221
222        # Do we have a magnetic variation? If so, provide mag winds,
223        # otherwise use an empty string.
224        if self.magnetic_variation is not None:
225            mag_winds = '%3.1f' % (self.true_wind_dir - self.magnetic_variation)
226        else:
227            mag_winds = ''
228
229        # Assemble string, compute checksum, and return it.
230        result_str = '%s,%3.1f,T,%s,M,%3.1f,N,%3.1f,M' % \
231                     (self.mwd_talker_id, self.true_wind_dir, mag_winds,
232                      self.true_wind_speed_kt, self.true_wind_speed_ms)
233        checksum = reduce(xor, (ord(c) for c in result_str))
234        return '$%s*%02X' % (result_str, checksum)

238"""Take in records and emit a NMEA XDR string, as per format:
239
240  $--XDR,a,x.x,a,c--c, ..... *hh<CR><LF> \\
241Field Number:
2421) Transducer Type
2432) Measurement Data
2443) Units of measurement
2454) Name of transducer
246x) More of the same
247n) Checksum
248Example:
249$IIXDR,C,19.52,C,TempAir*19
250$IIXDR,P,1.02481,B,Barometer*29
251Measured Value | Transducer Type | Measured Data   | Unit of measure | Transducer Name
252------------------------------------------------------------------------------------------------------
253barometric     | "P" pressure    | 0.8..1.1 or 800..1100           | "B" bar         | "Barometer"
254air temperature| "C" temperature |   2 decimals                    | "C" celsius     | "TempAir" or "ENV_OUTAIR_T"
255pitch          | "A" angle       |-180..0 nose down 0..180 nose up | "D" degrees     | "PTCH" or "PITCH"
256rolling        | "A" angle       |-180..0 L         0..180 R       | "D" degrees     | "ROLL"
257water temp     | "C" temperature |   2 decimals                    | "C" celsius     | "ENV_WATER_T"
258-----------------------------------------------------------------------------------------------------
259
260We're going to cheat a bit here, as traditionally, a Transform is only
261supposed to output zero or one record for every input record it
262gets. We're going to emit multiple records as separate lines in a
263single record and count on whatever gets them next (UDPWriter or
264TextFileWriter, for example) acting appropriately.
265"""
266################################################################################

269class XDRTransform:
270    """Output a NMEA XDR string, given whatever variables we can find.
271    """
272
273    def __init__(self, kwargs):
274        """
275        Look for these keys in the kwargs dict:
276        ```
277        barometer_field
278                 Name of field that contains barometric pressure.
279        barometer_output_field
280                 Transducer name of that should be output with barometer data.
281                 Defaults to barometer_field.
282        air_temp_field
283                 Name of field that contains air temperature
284        air_temp_output_field
285                 Transducer name of that should be output with air temp data.
286                 Defaults to air_temp_field.
287        sea_temp_field
288                 Name of field that contains water temperature
289        sea_temp_output_field
290                 Transducer name of that should be output with sea temp data.
291                 Defaults to sea_temp_field.
292        talker_id
293                 Should be format '--XDR' to identify the instrument
294                 that's creating the message.
295        ```
296        """
297        self.barometer_field = kwargs.get('barometer_field')
298        self.barometer_output_field = kwargs.get('barometer_output_field',
299                                                 self.barometer_field)
300        self.air_temp_field = kwargs.get('air_temp_field')
301        self.air_temp_output_field = kwargs.get('air_temp_output_field',
302                                                self.air_temp_field)
303        self.sea_temp_field = kwargs.get('sea_temp_field')
304        self.sea_temp_output_field = kwargs.get('sea_temp_output_field',
305                                                self.sea_temp_field)
306        self.xdr_talker_id = kwargs.get('xdr_talker_id')
307
308    ############################
309    def transform(self, record):
310        """Incorporate any useable fields in this record, and if it gives us a
311        new true wind value, return the results.
312        """
313        # Assume we have a single record; check that we've got the right
314        # record type.
315        if not record or type(record) is not dict:
316            logging.warning('Improper type for value dict: %s', type(record))
317            return None
318        fields = record.get('fields')
319        if not fields:
320            logging.debug('XDRTransform got record with no fields: %s', record)
321            return None
322
323        # Grab any relevant values
324        results = []
325        if self.barometer_field in fields:
326            barometer = fields.get(self.barometer_field)
327            barometer_data = '%s,P,%s,B,%s' % (self.xdr_talker_id, barometer,
328                                               self.barometer_output_field)
329            barometer_str = '$%s*%s' % (barometer_data, checksum(barometer_data))
330            results.append(barometer_str)
331
332        if self.air_temp_field in fields:
333            air_temp = fields.get(self.air_temp_field)
334            air_temp_data = '%s,C,%3.2f,C,%s' % (self.xdr_talker_id, float(air_temp),
335                                                 self.air_temp_output_field)
336            air_temp_str = '$%s*%s' % (air_temp_data, checksum(air_temp_data))
337            results.append(air_temp_str)
338
339        if self.sea_temp_field in fields:
340            sea_temp = fields.get(self.sea_temp_field)
341            sea_temp_data = '%s,C,%3.2f,C,%s' % (self.xdr_talker_id, float(sea_temp),
342                                                 self.sea_temp_output_field)
343            sea_temp_str = '$%s*%s' % (sea_temp_data, checksum(sea_temp_data))
344            results.append(sea_temp_str)
345
346        return results

350class DPTTransform:

372        dpt_talker_id
373                 Should be format '--DPT' to identify the instrument
374                 that's creating the message.
375        ```
376        """
377
378        self.depth_field = kwargs.get('depth_field')
379        self.offset_field = kwargs.get('offset_field')
380
381        self.dpt_talker_id = kwargs.get('dpt_talker_id')
382
383    ############################
384    def transform(self, record):
385        """Incorporate any useable fields in this record, and if it gives us a
386        new true wind value, return the results.
387        """
388        # Check that we've got the right record type - it should be a
389        # single record.
390        if not record or type(record) is not dict:
391            logging.warning('Improper type for record: %s', type(record))
392            return None
393        fields = record.get('fields')
394        if not fields:
395            logging.debug('MWDTransform got record with no fields: %s', record)
396            return None
397
398        depth = fields.get(self.depth_field)
399        offset = fields.get(self.offset_field)
400
401        if depth:
402            data = f'{self.dpt_talker_id},{depth},{offset}'
403            string = f'${data}*{checksum(data)}'
404            return string
405
406        return None
