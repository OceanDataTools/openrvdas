 21def is_string_or_list_of_strings(cmd):
 22    if isinstance(cmd, str):
 23        return True
 24    elif isinstance(cmd, list):
 25        return all(isinstance(item, str) for item in cmd)
 26    return False

 30def is_dict_of_lists_of_strings(cmd):
 31    if not isinstance(cmd, dict):
 32        return False
 33
 34    # So we have a dict. Check that all values are either strings
 35    # or lists of strings
 36    for value in cmd.values():
 37        if isinstance(value, str):
 38            continue
 39        elif (isinstance(value, list) and
 40              all(isinstance(item, str) for item in value)):
 41            continue
 42        else:
 43            return False
 44
 45    # If got this far, it all checks out
 46    return True

 70            strings, each string in the list is sent in sequence.

 74            or destroyed.

 77            Much as start_cmd and stop_cmd, except the string or list of strings
 78            are sent each time the PolledSerialReader's read() method is called,
 79            prior to trying to read from the port.
 80

 82            of lists of strings:

 86                  key2: ['command 2_1', 'command 2_2]
 87                  key3: ...
 88                ...
 89            The first time read() is called, the strings associated with key1 will be
 90            sent. The second time, those associated with key2, and so on. When the end
 91            of the dict is reached, it will start again with key1.
 92
 93        timeout <seconds>
 94            If timeout is specified, then the serial read will time out after this many
 95            seconds. It will then reissue the pre_read_cmds, if there is one, and try
 96            reading again. If there is a dict of pre_read_cmds, it will progress to the
 97            next one in the dict.
 98        ```

100        followed by a number (e.g. ``__PAUSE__ 5``), it will be interpreted as a command
101        to pause for that many seconds prior to sending the next command. If no number
102        is given, it will pause for one second.

148    def _send_command(self, cmd):
149        """Check if command is a 'pause'; if so, sleep, otherwise send it to serial port."""
150        logging.debug(f'Sending {cmd}')
151        if cmd.find('__PAUSE__') == 0:
152            pause_cmd = cmd.split()
153            if len(pause_cmd) == 1:
154                pause_length = 1
155            elif len(pause_cmd) == 2:
156                try:
157                    pause_length = float(pause_cmd[1])
158                except ValueError:
159                    raise ValueError('__PAUSE__ interval in PolledSerialReader must be '
160                                     'a float. Found: %s' % cmd)
161            else:
162                raise ValueError('Pause format "__PAUSE__ <seconds>"; found %s' % cmd)
163            logging.info('Pausing %g seconds', pause_length)
164            time.sleep(pause_length)
165        else:
166            # If it's a normal command we're sending
167            logging.info('Sending serial command "%s"', cmd)
168            self.serial.write(self._encode_str(cmd, unescape=True))
169            self.serial.flush()
170
171        logging.debug(f'Done sending {cmd}')

174    def read(self):
175        try:
176            # Try again if our read times out, or our serial port returns None
177            while True:
178                # Do we need to send anything prior to reading?
179                if not self.pre_read_cmd:  # no pre_read_cmd
180                    pass
181                elif isinstance(self.pre_read_cmd, list):  # list of commands
182                    for cmd in self.pre_read_cmd:
183                        self._send_command(cmd)
184                elif isinstance(self.pre_read_cmd, dict):  # dict of lists of commands
185                    key, commands = next(self.command_cycle)
186                    for cmd in commands:
187                        self._send_command(cmd)
188                elif self.pre_read_cmd:   # simple string command
189                    self._send_command(self.pre_read_cmd)
190
191                logging.debug('read() is being called')
192                record = super().read()
193                logging.debug(f'Returned from read() with: {record}')
194                if record is not None:
195                    return record
196                logging.debug('Serial read returned None; trying again.')
197
