openrvdas.logger.writers.grafana_live_writer

This writer pushes data to Grafana Live using the /api/live/push endpoint with InfluxDB Line Protocol format. This is a supported, stable API.

NOTE: This code was created substantially via Google Gemini. The usual AI code disclaimers apply - watch your topknot...

SECURITY - AUTHENTICATION TOKEN: The Grafana Service Account Token can be provided in three ways, checked in the following order of priority:

  1. token_file (BEST): Path to a file containing the token (e.g., /etc/secrets/grafana_token). Set file permissions to 600 so only the process owner can read it.

  2. GRAFANA_API_TOKEN (GOOD): Environment variable. Better than hardcoding, but visible to root.

  3. api_token (LEAST SECURE): Passed directly as a string argument. Discouraged for production use as it may appear in version control or process lists.

165    def _http_worker(self):
166        """Background worker that sends batched data to Grafana."""
167        # Batches is a dict mapping stream_id -> list of payload strings
168        # Example: { 'openrvdas/mru/rot': ['line1', 'line2'], 'openrvdas/pos/loc': ['line3'] }
169        batches = {}
170        pending_count = 0
171        last_send = time.time()
172
173        while not self.stop_event.is_set():
174            try:
175                # Get item with timeout to allow checking stop_event
176                try:
177                    # Queue items are (stream_id, payload_str)
178                    stream_id, payload_str = self.queue.get(timeout=1.0)
179
180                    if stream_id not in batches:
181                        batches[stream_id] = []
182
183                    batches[stream_id].append(payload_str)
184                    pending_count += 1
185                    self.queue.task_done()
186                except queue.Empty:
187                    pass
188
189                # Send batches if threshold reached or timeout elapsed
190                time_since_last = time.time() - last_send
191                should_send = (pending_count >= self.batch_size or
192                               (pending_count > 0 and time_since_last > 1.0))
193
194                if should_send:
195                    # Send each stream's batch separately
196                    for target_stream, batch_list in batches.items():
197                        if batch_list:
198                            self._send_batch(target_stream, batch_list)
199
200                    # Reset state
201                    batches = {}
202                    pending_count = 0
203                    last_send = time.time()
204
205            except Exception as e:
206                logging.error(f'Unexpected error in GrafanaLiveWriter worker: {e}')
207                # Don't tight loop on error
208                time.sleep(1)
209
210        # Send any remaining batches on shutdown
211        for target_stream, batch_list in batches.items():
212            if batch_list:
213                self._send_batch(target_stream, batch_list)

215    def _send_batch(self, stream_id, batch):
216        """Send a batch of line protocol records to a specific Grafana stream."""
217        safe_stream_id = quote(stream_id, safe='')
218        url = f"{self.base_api_url}/{safe_stream_id}"
219
220        try:
221            # Join batch with newlines
222            payload = '\n'.join(batch)
223            data_bytes = payload.encode('utf-8')
224
225            req = urllib.request.Request(url, data=data_bytes, method='POST')
226            req.add_header('Authorization', f'Bearer {self.api_token}')
227            req.add_header('Content-Type', 'text/plain')
228
229            with urllib.request.urlopen(req, timeout=5) as response:
230                if response.status == 204 or response.status == 200:
231                    with self.stats_lock:
232                        self.stats['sent'] += len(batch)
233                    logging.debug(f'Sent {len(batch)} records to Grafana stream {stream_id}')
234
235        except urllib.error.HTTPError as e:
236            error_msg = f'HTTP {e.code}: {e.reason}'
237            with self.stats_lock:
238                self.stats['errors'] += 1
239                self.stats['last_error'] = error_msg
240
241            logging.error(f'Grafana Push Error {error_msg} | URL: {url}')
242
243            # Read error body for details
244            try:
245                error_body = e.read().decode('utf-8')
246                logging.error(f'Error details: {error_body}')
247            except Exception:
248                pass
249
250            if e.code in (401, 403):
251                logging.error('Authentication error - check API token')
252                time.sleep(5)
253            elif e.code == 404:
254                logging.error('Endpoint not found - check Grafana version supports Live')
255                time.sleep(5)
256
257        except urllib.error.URLError as e:
258            error_msg = f'Connection error: {e.reason}'
259            with self.stats_lock:
260                self.stats['errors'] += 1
261                self.stats['last_error'] = error_msg
262
263            logging.error(f'Grafana Connection Error: {e.reason}')
264            time.sleep(1)
265
266        except Exception as e:
267            error_msg = str(e)
268            with self.stats_lock:
269                self.stats['errors'] += 1
270                self.stats['last_error'] = error_msg
271
272            logging.error(f'Unexpected error sending to Grafana: {e}')

274    def _normalize_record(self, record):
275        """Extract data_id, message_type, timestamp, and fields from record."""
276        if isinstance(record, str):
277            try:
278                record = DASRecord(json_str=record)
279            except json.JSONDecodeError:
280                return None, None, None, None
281
282        if isinstance(record, dict):
283            msg_type = record.get('message_type')
284            ts = record.get('timestamp', time.time())
285            did = record.get('data_id')
286            fields = record.get('fields', record)
287
288            return did, msg_type, ts, fields
289
290        if isinstance(record, DASRecord):
291            return record.data_id, record.message_type, record.timestamp, record.fields
292
293        return None, None, None, None

303    def _format_line_protocol(self, measurement, fields, timestamp, tags=None):
304        """
305        Format data in InfluxDB Line Protocol.
306
307        Format: measurement[,tag=value...] field=value[,field=value...] [timestamp]
308        """
309        if not fields:
310            return None
311
312        # Format fields
313        field_parts = []
314        for k, v in fields.items():
315            k_escaped = self._escape_key(k)
316
317            # Format value based on type
318            if isinstance(v, str):
319                v_escaped = v.replace('"', '\\"')
320                field_parts.append(f'{k_escaped}="{v_escaped}"')
321            elif isinstance(v, bool):
322                field_parts.append(f'{k_escaped}={str(v).upper()}')
323            elif isinstance(v, int):
324                field_parts.append(f'{k_escaped}={v}i')
325            elif isinstance(v, float):
326                field_parts.append(f'{k_escaped}={v}')
327            else:
328                v_escaped = str(v).replace('"', '\\"')
329                field_parts.append(f'{k_escaped}="{v_escaped}"')
330
331        if not field_parts:
332            return None
333
334        # Escape and format measurement name
335        measurement = self._escape_key(measurement.replace(' ', '_'))
336
337        # Format tags if provided
338        tag_str = ''
339        if tags:
340            tag_parts = [f'{self._escape_key(k)}={self._escape_tag_value(v)}'
341                         for k, v in tags.items()]
342            if tag_parts:
343                tag_str = ',' + ','.join(tag_parts)
344
345        # Format fields
346        field_str = ','.join(field_parts)
347
348        # Timestamp in nanoseconds
349        ts_ns = int(timestamp * 1e9)
350
351        return f'{measurement}{tag_str} {field_str} {ts_ns}'

353    def write(self, record: Union[DASRecord, dict, str]):
354        """
355        Write a record to Grafana Live.
356
357        The stream ID for the record is constructed as:
358            {base_stream}/{data_id}/{message_type}
359
360        Args:
361            record: DASRecord, dict, or JSON string
362        """
363        # See if it's something we can process, and if not, try digesting
364        if not self.can_process_record(record):  # inherited from BaseModule()
365            self.digest_record(record)  # inherited from BaseModule()
366            return
367
368        data_id, message_type, timestamp, fields = self._normalize_record(record)
369
370        if not fields:
371            return
372
373        # Determine measurement name (keep existing logic for Line Protocol)
374        measurement = message_type or self.measurement_name or data_id or 'default'
375
376        # Optional: Add data_id as a tag for better querying
377        tags = {'data_id': data_id} if data_id else None
378
379        # Format payload
380        payload = self._format_line_protocol(measurement, fields, timestamp, tags)
381
382        if payload:
383            # Honestly, this section is a bit voodoo. The current setup works
384            # both when there is a message_id and when there isn't, and I'm not
385            # sure why.
386
387            # Construct dynamic Stream ID
388            # Safe quoting of components to ensure valid URL structure
389            # Default to 'unknown' if parts are missing to avoid malformed URLs
390            safe_data_id = quote(str(data_id or 'unknown'), safe='')
391
392            # The message_id gets folded in by _format_line_protocol(), so
393            # don't need to duplicate (Gemini-written) code below
394            #safe_msg_type = quote(str(message_type or 'unknown'), safe='')
395            #target_stream_id = (f"{self.base_stream_id}/{safe_data_id}/"
396            #                    f"{safe_msg_type}")
397
398            # Without the logic below, we get duplicate data_ids when there's
399            # no message_id. I...don't understand, but I'm going with it.
400            if message_type:
401                target_stream_id = f"{self.base_stream_id}/{safe_data_id}"
402            else:
403                target_stream_id = f"{self.base_stream_id}"
404
405            try:
406                # Put tuple (stream_id, payload) into queue
407                self.queue.put_nowait((target_stream_id, payload))
408            except queue.Full:
409                with self.stats_lock:
410                    self.stats['dropped'] += 1
411                logging.warning('GrafanaLiveWriter queue full; dropping record')

304    def _format_line_protocol(self, measurement, fields, timestamp, tags=None):
305        """
306        Format data in InfluxDB Line Protocol.
307
308        Format: measurement[,tag=value...] field=value[,field=value...] [timestamp]
309        """
310        if not fields:
311            return None
312
313        # Format fields
314        field_parts = []
315        for k, v in fields.items():
316            k_escaped = self._escape_key(k)
317
318            # Format value based on type
319            if isinstance(v, str):
320                v_escaped = v.replace('"', '\\"')
321                field_parts.append(f'{k_escaped}="{v_escaped}"')
322            elif isinstance(v, bool):
323                field_parts.append(f'{k_escaped}={str(v).upper()}')
324            elif isinstance(v, int):
325                field_parts.append(f'{k_escaped}={v}i')
326            elif isinstance(v, float):
327                field_parts.append(f'{k_escaped}={v}')
328            else:
329                v_escaped = str(v).replace('"', '\\"')
330                field_parts.append(f'{k_escaped}="{v_escaped}"')
331
332        if not field_parts:
333            return None
334
335        # Escape and format measurement name
336        measurement = self._escape_key(measurement.replace(' ', '_'))
337
338        # Format tags if provided
339        tag_str = ''
340        if tags:
341            tag_parts = [f'{self._escape_key(k)}={self._escape_tag_value(v)}'
342                         for k, v in tags.items()]
343            if tag_parts:
344                tag_str = ',' + ','.join(tag_parts)
345
346        # Format fields
347        field_str = ','.join(field_parts)
348
349        # Timestamp in nanoseconds
350        ts_ns = int(timestamp * 1e9)
351
352        return f'{measurement}{tag_str} {field_str} {ts_ns}'

Grafana Live supports ingesting data via InfluxDB Line Protocol at: POST /api/live/push/{stream_id}

The 'stream_id' passed during initialization is treated as the 'base_stream'. Records are written to dynamic streams constructed as: {base_stream}/{data_id}/{message_type}

This allows a single Writer to populate multiple Grafana Live channels based on the content of the records.
