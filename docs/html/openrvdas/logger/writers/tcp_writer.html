 29

 32                     about failures.
 33

 35                     before sending
 36

 38                     you don't know you need this, don't enable it.
 39
 40        reuseport    Specifies wether to set SO_REUSEPORT on the created socket.  If
 41                     you don't know you need this, don't enable it.
 42
 43        encoding - 'utf-8' by default. If empty or None, do not attempt any
 44                decoding and return raw bytes. Other possible encodings are
 45                listed in online documentation here:
 46                https://docs.python.org/3/library/codecs.html#standard-encodings
 47
 48        encoding_errors - 'ignore' by default. Other error strategies are
 49                'strict', 'replace', and 'backslashreplace', described here:
 50                https://docs.python.org/3/howto/unicode.html#encodings
 51
 52        ```
 53        """
 54        super().__init__(**kwargs)  # processes 'quiet', encodings and type hints
 55
 56        self.num_retry = num_retry
 57        self.warning_limit = warning_limit
 58        self.num_warnings = 0
 59
 60        # 'eol' comes in as a (probably escaped) string. We need to
 61        # unescape it, which means converting to bytes and back.
 62        if eol is not None and self.encoding:
 63            eol = self._unescape_str(eol)
 64        self.eol = eol
 65
 66        # do name resolution once in the constructor
 67        #
 68        # NOTE: This means the hostname must be valid when we start, otherwise
 69        #       the config_check code will puke.  That's fine.  The alternative
 70        #       is we let name resolution happen while we're running, but then
 71        #       each failed lookup is going to block our write() routine for a
 72        #       few seconds - not good.
 73        #
 74        # NOTE: This also catches specifying impropperly formatted IP
 75        #       addresses.  The only way through gethostbyname() w/out throwing
 76        #       an exception is to provide a valid hostname or IP address.
 77        #       Propperly formatted IPs just get returned.
 78        #
 79        self.destination = socket.gethostbyname(destination)

 88        #
 89        # NOTE: Since connect() can actually fail w/ a TCP socket, we don't try
 90        #       that here.  Let's just do safe things.
 91        #
 92        self.socket = None

101    def _open_socket(self):
102        """Do socket prep so we're ready to write().  Returns socket object or None on
103        failure.
104        """
105        this_socket = socket.socket(family=socket.AF_INET,
106                                    type=socket.SOCK_STREAM,
107                                    proto=socket.IPPROTO_TCP)
108        if self.reuseaddr:
109            this_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, True)
110        if self.reuseport:
111            try:  # Raspbian doesn't recognize SO_REUSEPORT
112                this_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEPORT, True)
113            except AttributeError:
114                logging.warning('Unable to set socket REUSEPORT; may be unsupported')
115
116        # Try connecting
117        try:
118            this_socket.connect((self.destination, self.port))
119        except OSError as e:
120            if self.num_warnings < self.warning_limit:
121                logging.error('Unable to connect to %s:%d: %s', self.destination, self.port, e)
122                self.num_warnings += 1
123                if self.num_warnings == self.warning_limit:
124                    logging.error('TCPWriter._open_socket() - muting errors')
125            return None
126
127        # success, reset warning counter
128        if self.num_warnings == self.warning_limit:
129            logging.info('TCPWriter._open_socket() successfully connected after a series of '
130                         'failures; restting warnings.')
131        self.num_warnings = 0
132        return this_socket

143    def write(self, record: Union[str, bytes]):
144        """Write the record to the network."""
145
146        # See if it's something we can process, and if not, try digesting
147        if not self.can_process_record(record):  # inherited from BaseModule()
148            self.digest_record(record)  # inherited from BaseModule()
149            return
150
151        # Append eol if configured
152        if self.eol:
153            record += self.eol
154
155        # NOTE: Unlike UDP socket, which really only can detect failure during
156        #       send() (and even then only very poorly), a TCP connect() can
157        #       fail, so we need to track attempts to connect in order to honor
158        #       `num_retry` on a disconnected socket.
159        #
160        #       We also need to tear down and start over with a fresh connect()
161        #       if send() fails.
162        #
163        num_tries = 0
164        bytes_sent = 0
165        rec_len = len(record)
166        while num_tries <= self.num_retry and bytes_sent < rec_len:
167            num_tries += 1
168            # attempt to connect socket if needed, up to `num_tries` times
169            if not self.socket:
170                self.socket = self._open_socket()
171            if not self.socket:
172                # no need for further error messages, _open_socket() will have
173                # already complained sufficiently
174                continue
175
176            # attempt to verify we're really connected
177            #
178            # NOTE: This is because the only way to detect a closed TCP socket
179            #       is via recv(), which is awkward when we're trying to
180            #       guarantee a write was successful.
181            #
182            #         https://stackoverflow.com/questions/49457631/c-server-socket-closed-but-client-socket-still-able-send-two-more-packages
183            #
184            #         "send() just puts the data in the kernel socket buffer,
185            #         it doesn't wait for the data to be transmitted or the
186            #         server to acknowledge receipt of it.
187            #
188            #         You don't get SIGPIPE until the data is transmitted and
189            #         the server rejects it by sending a RST segment.
190            #
191            #         It works this way because each direction of a TCP
192            #         connection is treated independently. When the server
193            #         closes the socket, it sends a FIN segment. This just
194            #         tells the client that the server is done sending data, it
195            #         doesn't mean that the server cannot receive data. There's
196            #         nothing in the TCP protocol that allows the server to
197            #         inform the client of this. So the only way to find out
198            #         that it's not accepting any more data is when the client
199            #         gets that RST response.
200            #
201            #         Informing the client that they shouldn't send anything
202            #         more is usually done in the application protocol, since
203            #         it's not available in TCP."
204            #
205            #       So it might take a couple "successful" send() calls before
206            #       the socket layer notices nobody's home on the other end.
207            #
208            #       Fine then, let's do a 1 byte recv() w/ flags arranged so we
209            #       don't really read off the socket forcing it to raise
210            #       BlockingIOError if the socket is still connected.  Not
211            #       perfect, because the remote side could still close after
212            #       this call but before our send() below, but it's something.
213            #
214            try:
215                # If the socket is still connected, this will raise
216                # BlockingIOError.  If it's not, it returns 0 bytes.
217                msg = self.socket.recv(1, socket.MSG_DONTWAIT | socket.MSG_PEEK)  # noqa F841
218                logging.error('TCPWriter: connection closed')
219                self.num_warnings += 1
220                self._close_socket(self.socket)
221                self.socket = None
222                continue
223            except BlockingIOError:
224                pass
225
226            # we're connected, try sending
227            try:
228                bytes_sent = self.socket.send(self._encode_str(record))
229            except OSError as e:
230                # send failed, we need to disconnect and start over
231                #
232                # NOTE: In order to get this far, we have to have successfully
233                #       connected, which means we JUST reset self.num_warnings
234                #
235                logging.error('TCPWriter: send() error: %s:%d: %s',
236                              self.destination, self.port, str(e))
237                self.num_warnings += 1
238                self._close_socket(self.socket)
239                self.socket = None
240                continue
241
242            # check to see if we really wrote it all
243            if bytes_sent < rec_len and not self.quiet:
244                logging.warning('TCPWriter: send() did not send the whole record: '
245                                'bytes_sent=%d, rec_len=%d', bytes_sent, rec_len)
246
247        logging.debug('TCPWriter.write() wrote %d/%d bytes after %d tries',
248                      bytes_sent, rec_len, num_tries)

174                # already complained sufficiently
175                continue
176
177            # attempt to verify we're really connected
178            #
179            # NOTE: This is because the only way to detect a closed TCP socket
180            #       is via recv(), which is awkward when we're trying to
181            #       guarantee a write was successful.
182            #
183            #         https://stackoverflow.com/questions/49457631/c-server-socket-closed-but-client-socket-still-able-send-two-more-packages
184            #
185            #         "send() just puts the data in the kernel socket buffer,
186            #         it doesn't wait for the data to be transmitted or the
187            #         server to acknowledge receipt of it.
188            #
189            #         You don't get SIGPIPE until the data is transmitted and
190            #         the server rejects it by sending a RST segment.
191            #
192            #         It works this way because each direction of a TCP
193            #         connection is treated independently. When the server
194            #         closes the socket, it sends a FIN segment. This just
195            #         tells the client that the server is done sending data, it
196            #         doesn't mean that the server cannot receive data. There's
197            #         nothing in the TCP protocol that allows the server to
198            #         inform the client of this. So the only way to find out
199            #         that it's not accepting any more data is when the client
200            #         gets that RST response.
201            #
202            #         Informing the client that they shouldn't send anything
203            #         more is usually done in the application protocol, since
204            #         it's not available in TCP."
205            #
206            #       So it might take a couple "successful" send() calls before
207            #       the socket layer notices nobody's home on the other end.
208            #
209            #       Fine then, let's do a 1 byte recv() w/ flags arranged so we
210            #       don't really read off the socket forcing it to raise
211            #       BlockingIOError if the socket is still connected.  Not
212            #       perfect, because the remote side could still close after
213            #       this call but before our send() below, but it's something.
214            #
215            try:
216                # If the socket is still connected, this will raise
217                # BlockingIOError.  If it's not, it returns 0 bytes.
218                msg = self.socket.recv(1, socket.MSG_DONTWAIT | socket.MSG_PEEK)  # noqa F841
219                logging.error('TCPWriter: connection closed')
220                self.num_warnings += 1
221                self._close_socket(self.socket)
222                self.socket = None
223                continue
224            except BlockingIOError:
225                pass
226
227            # we're connected, try sending
228            try:
229                bytes_sent = self.socket.send(self._encode_str(record))
230            except OSError as e:
231                # send failed, we need to disconnect and start over
232                #
233                # NOTE: In order to get this far, we have to have successfully
234                #       connected, which means we JUST reset self.num_warnings
235                #
236                logging.error('TCPWriter: send() error: %s:%d: %s',
237                              self.destination, self.port, str(e))
238                self.num_warnings += 1
239                self._close_socket(self.socket)
240                self.socket = None
241                continue
242
243            # check to see if we really wrote it all
244            if bytes_sent < rec_len and not self.quiet:
245                logging.warning('TCPWriter: send() did not send the whole record: '
246                                'bytes_sent=%d, rec_len=%d', bytes_sent, rec_len)
247
248        logging.debug('TCPWriter.write() wrote %d/%d bytes after %d tries',
249                      bytes_sent, rec_len, num_tries)
