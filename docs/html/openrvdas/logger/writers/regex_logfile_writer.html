 15    """Write to the specified filebase, with datestamp appended. If filebase
 16    is a <regex>:<filebase> dict, write records to every filebase whose
 17    regex appears in the record.
 18    """
 19    def __init__(self, filebase=None, flush=True,
 20                 time_format=timestamp.TIME_FORMAT,
 21                 date_format=timestamp.DATE_FORMAT,
 22                 split_char=' ', suffix='', header=None,
 23                 header_file=None, rollover_hourly=False,
 24                 **kwargs):
 25        """Write timestamped text records to a filebase. The filebase will
 26        have the current date appended, in keeping with R2R format
 27        recommendations (http://www.rvdata.us/operators/directory). When the
 28        timestamped date on records rolls over to next day, create a new file
 29        with the new date suffix.
 30
 31        If filebase is a dict of <string>:<filebase> pairs, The writer will
 32        attempt to match a <string> in the dict to each record it receives.
 33        It will write the record to the filebase corresponding to the first
 34        string it matches (Note that the order of comparison is not
 35        guaranteed!). If no strings match, the record will be written to the
 36        standalone filebase provided.
 37        ```
 38        filebase        A filebase string to write to or a dict mapping
 39                        <string>:<filebase>.
 40
 41        flush           If True (default), flush after every write() call
 42
 43        date_fomat      A strftime-compatible string, such as '%Y-%m-%d';
 44                        defaults to whatever's defined in
 45                        utils.timestamps.DATE_FORMAT.
 46
 47        split_char      Delimiter between timestamp and rest of message
 48
 49        suffix          string to apply to the end of the log filename
 50
 51        header          Add the specified header string to each file.
 52
 53        header_file     Add the content of the specified file to each file.
 54
 55        rollover_hourly Set files to truncate by hour.  By default files will
 56                        truncate by day
 57
 58        quiet           If True, don't complain if a record doesn't match
 59                        any mapped prefix
 60        ```
 61        """
 62        super().__init__(**kwargs)  # processes 'quiet' and type hints
 63
 64        self.filebase = filebase
 65        self.flush = flush
 66        self.time_format = time_format
 67        self.date_format = date_format
 68        self.split_char = split_char
 69        self.suffix = suffix
 70        self.header = header
 71        self.header_file = header_file
 72        self.rollover_hourly = rollover_hourly
 73
 74        # If our filebase is a dict, we're going to be doing our
 75        # fancy pattern->filebase mapping.
 76        self.do_filebase_mapping = isinstance(self.filebase, dict)
 77
 78        if self.do_filebase_mapping:
 79            # Do our matches faster by precompiling
 80            self.compiled_filebase_map = {
 81                pattern: re.compile(pattern) for pattern in self.filebase
 82            }
 83        self.current_filename = {}
 84        self.writer = {}
 85
 86    ############################
 87    def write(self, record: str):
 88        """Note: Assume record begins with a timestamp string."""
 89
 90        # See if it's something we can process, and if not, try digesting
 91        if not self.can_process_record(record):  # inherited from BaseModule()
 92            self.digest_record(record)  # inherited from BaseModule()
 93            return
 94
 95        # Get the timestamp we'll be using
 96        try:  # Try to extract timestamp from record
 97            time_str = record.split(self.split_char)[0]
 98            ts = timestamp.timestamp(time_str, time_format=self.time_format)
 99        except ValueError:
100            if not self.quiet:
101                logging.error('LogfileWriter.write() - bad timestamp: "%s"', record)
102                return
103
104        # Now parse ts into hour and date strings
105        hr_str = self.rollover_hourly and \
106            timestamp.date_str(ts, date_format='_%H00') or ""
107        date_str = timestamp.date_str(ts, date_format=self.date_format)
108        time_str = date_str + hr_str + self.suffix
109        logging.debug('LogfileWriter time_str: %s', time_str)
110
111        # Figure out where we're going to write
112        if self.do_filebase_mapping:
113            matched_patterns = [self.write_if_match(record, pattern, time_str)
114                                for pattern in self.filebase]
115            if True not in matched_patterns:
116                if not self.quiet:
117                    logging.warning(f'No patterns matched in PatternLogfileWriter '
118                                    f'for record "{record}"')
119        else:
120            pattern = 'fixed'  # just an arbitrary fixed pattern
121            filename = self.filebase + '-' + time_str
122            self.write_filename(record, pattern, filename)
123
124    ############################
125    def write_filename(self, record, pattern, filename):
126        """Write record to filename. If it's the first time we're writing to
127        this filename, create the appropriate FileWriter and insert it into
128        the map for the relevant pattern."""
129
130        # Are we currently writing to this file? If not, open/create it.
131        if not filename == self.current_filename.get(pattern):
132            logging.info('LogfileWriter opening new file: %s', filename)
133            self.current_filename[pattern] = filename
134            self.writer[pattern] = FileWriter(filename=filename,
135                                              header=self.header,
136                                              header_file=self.header_file,
137                                              flush=self.flush)
138        # Now, if our logic is correct, should *always* have a matching_writer
139        matching_writer = self.writer.get(pattern)
140        matching_writer.write(record)
141
142    ############################
143    def write_if_match(self, record, pattern, time_str):
144        """If the record matches the pattern, write to the matching filebase."""
145        # Find the compiled regex matching the pattern
146        regex = self.compiled_filebase_map.get(pattern)
147        if not regex:
148            logging.error(f'System error: found no regex pattern matching "{pattern}"!')
149            return None
150
151        # If the pattern isn't in this record, go home quietly
152        if regex.search(record) is None:
153            return None
154
155        # Otherwise, we write.
156        filebase = self.filebase.get(pattern)
157        if filebase is None:
158            logging.error(f'System error: found no filebase matching pattern "{pattern}"!')
159            return None
160
161        filename = filebase + '-' + time_str
162        self.write_filename(record, pattern, filename)
163        return True
